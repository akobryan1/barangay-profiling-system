-- Create Database
CREATE DATABASE IF NOT EXISTS barangay_profiling;
USE barangay_profiling;

-- Table: Barangay
CREATE TABLE Barangay (
    barangay_id INT AUTO_INCREMENT PRIMARY KEY,
    barangay_name VARCHAR(100) NOT NULL,
    city VARCHAR(100),
    province VARCHAR(100),
    region VARCHAR(100)
);

-- Table: Family
CREATE TABLE Family (
    family_id INT AUTO_INCREMENT PRIMARY KEY,
    household_number VARCHAR(50),
    address VARCHAR(100),
    family_head_id INT,
    barangay_id INT,
    family_name VARCHAR(50),
    FOREIGN KEY (barangay_id) REFERENCES Barangay(barangay_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

-- Table: Person
CREATE TABLE Person (
    person_id INT AUTO_INCREMENT PRIMARY KEY,
    family_id INT,
    first_name VARCHAR(50),
    middle_name VARCHAR(50),
    last_name VARCHAR(50),
    gender VARCHAR(10),
    date_of_birth DATE,
    relationship_to_head VARCHAR(50),
    occupation VARCHAR(50),
    educational_attainment VARCHAR(50),
    contact_number VARCHAR(15),
    civil_status VARCHAR(50),
    religion VARCHAR(50),
    FOREIGN KEY (family_id) REFERENCES Family(family_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

-- Table: Barangay_Official
CREATE TABLE Barangay_Official (
    official_id INT AUTO_INCREMENT PRIMARY KEY,
    barangay_id INT,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    position VARCHAR(50),
    term_start DATE,
    term_end DATE,
    FOREIGN KEY (barangay_id) REFERENCES Barangay(barangay_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

-- Table: Barangay_Clearance
CREATE TABLE Barangay_Clearance (
    clearance_id INT AUTO_INCREMENT PRIMARY KEY,
    person_id INT,
    clearance_type VARCHAR(50),
    issue_date DATE,
    validity_date DATE,
    FOREIGN KEY (person_id) REFERENCES Person(person_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);
